-- kwerendy.txt — zapytania SQL dla bazy 'piekarnia'

-- KWERENDA 1
SELECT Rodzaj, Nazwa, Gramatura, Cena
FROM wyroby
WHERE Rodzaj = 'INNE';

-- KWERENDA 2
SELECT DISTINCT Rodzaj
FROM wyroby
ORDER BY Rodzaj DESC;

-- KWERENDA 3
SELECT ID, Nazwa
FROM wyroby
WHERE Nazwa LIKE '%Chałka%';

-- KWERENDA 4
SELECT Rodzaj, ROUND(AVG(Cena), 2) AS `Średnia cena`
FROM wyroby
GROUP BY Rodzaj;
